Leetcode65 有效数字

题目来源：leetcode 65

有效数字（按顺序）可以分成以下几个部分： 一个 小数 或者 整数 （可选）一个 ‘e’ 或 ‘E’ ，后面跟着一个 整数 小数（按顺序）可以分成以下几个部分： （可选）一个符号字符（‘+’ 或 ‘-'） 下述格式之一： 至少一位数字，后面跟着一个点 ‘.’ 至少一位数字，后面跟着一个点 ‘.’ ，后面再跟着至少一位数字 一个点 ‘.’ ，后面跟着至少一位数字 整数（按顺序）可以分成以下几个部分： （可选）一个符号字符（‘+’ 或 ‘-'） 至少一位数字 部分有效数字列举如下：[“2”, “0089”, “-0.1”, “+3.14”, “4.", “-.9”, “2e10”, “-90E3”, “3e+7”, “+6e-1”, “53.5e93”, “-123.456e789”] 部分无效数字列举如下：[“abc”, “1a”, “1e”, “e3”, “99e2.5”, “–6”, “-+3”, “95a54e53”] 给你一个字符串 s ，如果 s 是一个 有效数字 ，请返回 true 。 来源：力扣（LeetCode） 链接：https://leetcode-cn.com/problems/valid-number 著作权归领扣网络所有。商业转载请联系官方授权，非商业转载请注明出处。

看上去就是字符串匹配的题目，如果使用各种语言的正则表达式库则秒解决，而正则表达式的原理就是状态机，这里我也写了一个最原始版本的有限状态机。

class Solution {
public:
  enum STATE_TYPE {
    START_STATE,
    INT_SIGN_STATE,
    INT_STATE,
    POINT_STATE,
    NO_INT_POINT_STATE,
    FRACTION_STATE,
    EXP_STATE,
    EXP_SIGN_STATE,
    EXP_INT_STATE,
    END_STATE
  };
  enum CHAR_TYPE { SIGN_CHAR, INT_CHAR, EXP_CHAR, POINT_CHAR, OTHER_CHAR };
  CHAR_TYPE getCharType(char ch) {
    if (ch >= '0' && ch <= '9') {
      return INT_CHAR;
    }
    if (ch == '.') {
      return POINT_CHAR;
    }
    if (ch == 'e' || ch == 'E') {
      return EXP_CHAR;
    }
    if (ch == '+' || ch == '-') {
      return SIGN_CHAR;
    }
    return OTHER_CHAR;
  }
  bool isNumber(string s) {
    unordered_map<STATE_TYPE, unordered_map<CHAR_TYPE, STATE_TYPE>> stateChart{
        {START_STATE,
         {{SIGN_CHAR, INT_SIGN_STATE},
          {POINT_CHAR, NO_INT_POINT_STATE},
          {INT_CHAR, INT_STATE}}},
        {INT_SIGN_STATE,
         {{POINT_CHAR, NO_INT_POINT_STATE}, {INT_CHAR, INT_STATE}}},
        {NO_INT_POINT_STATE, {{INT_CHAR, FRACTION_STATE}}},
        {INT_STATE,
         {{POINT_CHAR, POINT_STATE},
          {INT_CHAR, INT_STATE},
          {EXP_CHAR, EXP_STATE}}},
        {POINT_STATE, {{INT_CHAR, FRACTION_STATE}, {EXP_CHAR, EXP_STATE}}},
        {FRACTION_STATE, {{INT_CHAR, FRACTION_STATE}, {EXP_CHAR, EXP_STATE}}},
        {EXP_STATE, {{SIGN_CHAR, EXP_SIGN_STATE}, {INT_CHAR, EXP_INT_STATE}}},
        {EXP_SIGN_STATE, {{INT_CHAR, EXP_INT_STATE}}},
        {EXP_INT_STATE, {{INT_CHAR, EXP_INT_STATE}}}};
    int n = s.length();
    STATE_TYPE state = START_STATE;
    for (int i = 0; i < n; i++) {
      CHAR_TYPE c = getCharType(s[i]);
      if (stateChart[state].find(c) == stateChart[state].end()) {
        return false;
      }
      state = stateChart[state][c];
    }
    return state == FRACTION_STATE || state == INT_STATE ||
           state == EXP_INT_STATE || state == POINT_STATE;
  }
};

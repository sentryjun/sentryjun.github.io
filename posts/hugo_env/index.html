关于搭建一个博客的想法，其实已经很久了，但是因为自己太懒就一直没有搞。最近刚好双十一看了一下自己再不薅点羊毛就要脱离学生优惠的群体了，就搞了个服务器使用Hugo搭建静态站点发布到githubPages的方案。

工作流程

整个方案的大概流程如下：

  1. 在我的台式机上编辑

1 Hugo安装

环境搭建

  • vscode
  • Ubuntu 20
  • Github
  • golang

hugo直接apt-get，在home下面创建博客

apt-get install hugo 
hugo new site myBlog

找了一堆主题，感觉LoveIt比较适合我，安装。

git init
git submodule add https://github.com/dillonzq/LoveIt.git themes/LoveIt

稍微配一下文件

baseURL = "http://example.org/"
# [en, zh-cn, fr, ...] 设置默认的语言
defaultContentLanguage = "zh-cn"
# 网站语言, 仅在这里 CN 大写
languageCode = "zh-CN"
# 是否包括中日韩文字
hasCJKLanguage = true
# 网站标题
title = "Jun的摸鱼老巢"

# 更改使用 Hugo 构建网站时使用的默认主题
theme = "LoveIt"

[params]
  # LoveIt 主题版本
  version = "0.2.X"

[menu]
  [[menu.main]]
    identifier = "posts"
    # 你可以在名称 (允许 HTML 格式) 之前添加其他信息, 例如图标
    pre = ""
    # 你可以在名称 (允许 HTML 格式) 之后添加其他信息, 例如图标
    post = ""
    name = "文章"
    url = "/posts/"
    # 当你将鼠标悬停在此菜单链接上时, 将显示的标题
    title = ""
    weight = 1
  [[menu.main]]
    identifier = "tags"
    pre = ""
    post = ""
    name = "标签"
    url = "/tags/"
    title = ""
    weight = 2
  [[menu.main]]
    identifier = "categories"
    pre = ""
    post = ""
    name = "分类"
    url = "/categories/"
    title = ""
    weight = 3

# Hugo 解析文档的配置
[markup]
  # 语法高亮设置 (https://gohugo.io/content-management/syntax-highlighting)
  [markup.highlight]
    # false 是必要的设置 (https://github.com/dillonzq/LoveIt/issues/158)
    noClasses = false

[params.social]
    github = "sentryjun"

[params.home.profile]
      enable = true
      # Gravatar 邮箱，用于优先在主页显示的头像
      gravatarEmail = ""
      # 主页显示头像的 URL
      avatarURL = "/images/avatar.png"
      # LoveIt 更改 | 0.2.7 主页显示的网站标题 (支持 HTML 格式)
      title = "Jun的摸鱼老巢"
      # 主页显示的网站副标题
      subtitle = "Code, Life, Sport and Thinking"
      # 是否为副标题显示打字机动画
      typeit = true
      # 是否显示社交账号
      social = true
